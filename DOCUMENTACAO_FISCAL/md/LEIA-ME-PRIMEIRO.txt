🎯 EXEMPLO RÁPIDO
==================

import { useState } from 'react';
import { ModalCancelarNfce } from '@/modules/fiscal';

export function MinhaTabela() {
  const [modal, setModal] = useState(false);
  const [nota, setNota] = useState(null);

  return (
    <>
      <table>
        <tbody>
          <tr>
            <td>1234</td>
            <td>
              <button onClick={() => {
                setNota({ id: 1, numero: '1234', chaveAcesso: 'xxx' });
                setModal(true);
              }}>
                Cancelar
              </button>
            </td>
          </tr>
        </tbody>
      </table>

      <ModalCancelarNfce
        nota={nota}
        isOpen={modal}
        onClose={() => setModal(false)}
        onSuccess={() => console.log('✅ Cancelado!')}
      />
    </>
  );
}
